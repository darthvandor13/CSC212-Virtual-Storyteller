  objective = _identity 
) 
 Select result of rules that minimizes objective

>>> from sympy.strategies import minimize
>>> inc = lambda x: x + 1
>>> dec = lambda x: x - 1
>>> rl = minimize(inc, dec)
>>> rl(4)
3

>>> rl = minimize(inc, dec, objective=lambda x: -x)  # maximize
>>> rl(4)
5
